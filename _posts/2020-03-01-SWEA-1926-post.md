* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <iostream>
#include <algorithm>

int main(int argc, char** argv) {
    int N;
    std::cin >> N;

    for(int i = 1; i <= N; i++) {
        std::string temp = std::to_string(i);
        int cnt = std::count_if(temp.begin(), temp.end(), [](char c) {
            return c == '3' || c == '6' || c == '9';
        });
        if(cnt != 0) {
            for(int j = 0; j < cnt; j++) {
                std::cout << "-";
            }
        } else {
            std::cout << i;
        }
        std::cout << " ";
    }
    std::cout << std::endl;
    return 0;
}
```
