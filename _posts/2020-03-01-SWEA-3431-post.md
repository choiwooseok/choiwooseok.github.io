* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <iostream>

int main(int argc, char** argv) {
    int T;
    std::cin >> T;
    for(int i = 0; i < T; i++) {
        int L, U, X;
        std::cin >> L >> U >> X;
        if( U <= X  ) {
            std::cout << "#" << i + 1 << " " << -1 << std::endl;
        } else if( X <= L) {
            std::cout << "#" << i + 1 << " " << L - X << std::endl;
        } else {
            std::cout << "#" << i + 1 << " " << 0 << std::endl;            
        }
    }
    return 0;
}
```
