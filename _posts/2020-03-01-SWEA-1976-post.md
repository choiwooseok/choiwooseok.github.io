* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <iostream>

int main(int argc, char** argv) {
    int T;
    std::cin >> T;
    for(int i = 0; i < T; i++) {
        int h_fir, h_sec, m_fir, m_sec;
        std::cin >> h_fir >> m_fir >> h_sec >> m_sec;
        int h_sum = h_fir + h_sec;
        int m_sum = m_fir + m_sec;
        if(m_sum > 60) {
            m_sum -= 60;
            h_sum += 1;
        }
        if(h_sum > 12) {
            h_sum -= 12;
        }
        std::cout << "#" << i + 1 << " " << h_sum << " " << m_sum << std::endl;
    }

    return 0;
}
```
