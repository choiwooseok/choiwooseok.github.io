* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <iostream>

int main(int argc, char** argv) {
    int T;
    std::cin >> T;
    for(int i = 0; i < T; i++) {
        int sum = 0;
        int N;
        std::cin >> N;
        for(int j = 1; j <= N; j++) {
            if( j % 2 == 0) {
                sum -= j;
            } else {
                sum += j;
            }
        }
        std::cout << "#" << i + 1 << " " << sum << std::endl;
    }
    return 0;
}
```
