* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <iostream>
#include <unordered_set>
int main(int argc, char** argv) {
    int T;
    std::cin >> T;
    for(int i = 0; i < T; i++) {
        int x;
        std::cin >> x;
        std::unordered_set<int> temp;
        while(x > 0) {
            temp.insert( x % 10);
            x /= 10;
        }
        std::cout << "#" << i + 1 << " " << temp.size() << std::endl;
    }
    return 0;
}
```
