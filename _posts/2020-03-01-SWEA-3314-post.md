* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <iostream>
#include <array>

int main(int argc, char** argv) {
    int T;
    std::cin >> T;
    for(int i = 0; i < T; i++) {
        std::array<int, 5> scores;
        for(int j = 0; j < 5; j++) {
            std::cin >> scores[j];
        }
        int sum = 0;
        for(auto score : scores) {
            if(score < 40) {
                sum += 40;
            } else {
                sum += score;
            }
        }
        std::cout << "#" << i + 1 << " " << sum / 5 << std::endl;
    }
    return 0;
}
```
