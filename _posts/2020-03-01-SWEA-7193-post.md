* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <iostream>
#include <string>

long long digitSum(const std::string& X) {
    long long ret = 0LL;
    for(char ch : X) {
        ret += ch - '0';
    }
    return ret;
}

int main(int argc, char** argv) {
    int T;
    std::cin >> T;
    for(int i = 0; i < T; i++) {
        int N;
        std::string X;
        std::cin >> N >> X;
        std::cout << "#" << i + 1 << " " << digitSum(X) % (N-1) << std::endl;
    }
    return 0;
}
```
