* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <iostream>
#include <string>
#include <set>

void digitCount(int n, int i) {
    std::set<int> counted;
    std::string nStr = "";
    int cnt = 1;
    while(counted.size() != 10) {
        nStr = std::to_string(cnt * n);
        for(auto nChar : nStr) {
            counted.insert(nChar);
        }
        cnt++;
    }
    std::cout << "#" << i << " " << nStr << "\n";
}

int main(int argc, char** argv) {
    int T;
    std::cin >> T;
    for(int i = 1; i <= T; i++) {
        int N;
        std::cin >> N;
        digitCount(N , i);
    }
    return 0;
}
```
