* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <iostream>

int main() {
    int t;
    std::cin >> t;
    for(int i=1; i <=t; ++i) {
        int n;
        std::cin >> n;

        int a = 0,b = 0,c = 0, d = 0, e = 0;

        while(n % 2 == 0) {
            n/=2;
            a++;
        }
        while(n % 3 == 0) {
            n/=3;
            b++;
        }
        while(n % 5 == 0) {
            n/=5;
            c++;
        }
        while(n % 7 == 0) {
            n/=7;
            d++;
        }
        while(n % 11 == 0) {
            n/=11;
            e++;
        }
        std::cout << "#" << i << " " << a << " " << b << " " << c << " " << d << " " << e << "\n";
    }
    return 0;
}
```
