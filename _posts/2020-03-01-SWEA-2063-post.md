* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

int main() {
	int n;
    std::cin >> n;
    std::vector<int> inputs(n);
    for(int i = 0; i < n; i++) {
    	std::cin >> inputs[i];
    }
    std::sort(inputs.begin(), inputs.end());
    std::cout << inputs[std::round(inputs.size()/2)];
    return 0;
}
```
