* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <vector>
#include <iostream>
#include <numeric>

void oddSum() {
    int numCase = 0;
    std::cin >> numCase;
    for(int i = 0; i < numCase; i++) {
        std::vector<int> testCase(10, 0);
        for(int j = 0; j < 10; j++) {
            std::cin >> testCase[j];
        }
        std::cout << "#" << i + 1 << " "
            << std::accumulate(testCase.begin(), testCase.end(), 0,
                               [](int before, int n) {
                                   return n % 2 == 0 ? before : before + n;
                               })
            << "\n";
    }
}
int main() {
    oddSum();
    return 0;
}  
```
