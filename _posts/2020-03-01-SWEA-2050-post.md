* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <iostream>
#include <string>
#include <algorithm>

void answer(std::string s) {
    std::transform(s.begin(), s.end(), s.begin(), [](char c) {
        return std::tolower(c);
    });
    for(char c : s) {
    	std::cout << (int) c - 96 << " ";
    }
}

int main() {
    std::string str;
    std::cin >> str;
    answer(str);
	return 0;
}
}
```
