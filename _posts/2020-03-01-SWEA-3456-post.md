* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <iostream>

int main(int argc, char** argv) {
    int T;
    std::cin >> T;

    for(int i = 0; i < T; i++) {
        int a, b, c;
        std::cin >> a >> b >> c;
        if( a == b ) {
            std::cout << "#" << i + 1 << " " << c << std::endl;
        } else if (a == c) {
            std::cout << "#" << i + 1 << " " << b << std::endl;
        } else {
            std::cout << "#" << i + 1 << " " << a << std::endl;
        }
    }
    return 0;
}
```
