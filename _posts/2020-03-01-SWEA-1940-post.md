* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <iostream>

int main(int argc, char **argv) {
    int T;
    std::cin >> T;
    for (int i = 0; i < T; i++) {
        int N;
        std::cin >> N;
        int result = 0;
        int v = 0;

        for (int j = 0; j < N; j++) {
            int cmd;
            std::cin >> cmd;
            int a;
            switch (cmd) {
            case 0:
                result += v;
                break;
            case 1:
                std::cin >> a;
                v += a;
                result += v;
                break;
            case 2:
                std::cin >> a;
                if(v < a) {
                    v = 0;
                } else {
                    v -= a;
                }
                result += v;
                break;
            default:
                break;
            }
        }

        std::cout << "#" << i + 1 << " " << result << std::endl;
    }

    return 0;
}
```
