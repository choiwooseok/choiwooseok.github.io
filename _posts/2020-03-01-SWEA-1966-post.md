* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    int t;
    std::cin >> t;
    for(int i=1; i<=t; ++i) {
	    int n;
    	std::cin >> n;
        std::vector<int> v(n);
        for(int j=0; j < n; j++) {
        	std::cin >> v[j];
        }
        std::sort(v.begin(), v.end());
    	std::cout << "#" << i << " ";
        for(int num : v) {
        	std::cout << num << " ";
        }
        std::cout << "\n";
    }
	return 0;
}
```
