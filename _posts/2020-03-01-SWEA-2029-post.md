* 개인적인 풀이. 문제시 삭제 예정, 더 좋은 방법 공유 환영

```cpp
#include <iostream>

int main() {
	int n;
    std::cin >> n;
    for(int i=1; i <=n; ++i) {
    	int a, b;
        std::cin >> a >> b;
        std::cout << "#" << i << " " << a/b << " " << a%b << std::endl;
    }
	return 0;
}
```
